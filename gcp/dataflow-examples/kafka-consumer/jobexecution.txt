   -Dexec.mainClass=package.executor_consumer \

     --tempLocation=gs://bucket \

     --jobName=etl-kafka-consumer \

     --maxNumWorkers=20 \
     --diskSizeGb=100 \

     --workerMachineType=n1-standard-1 "
